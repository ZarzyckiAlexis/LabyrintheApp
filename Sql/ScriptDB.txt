CREATE TABLE jeu(
   ID_JEU SMALLINT AUTO_INCREMENT,
   NOM_DU_JEU VARCHAR(51),
   PRIMARY KEY(ID_JEU)
);

CREATE TABLE joueur(
   ID_JOUEUR SMALLINT AUTO_INCREMENT,
   NOM_DU_JOUEUR VARCHAR(101),
   PRIMARY KEY(ID_JOUEUR),
   UNIQUE(NOM_DU_JOUEUR)
);

CREATE TABLE partie(
   ID_PARTIE SMALLINT AUTO_INCREMENT,
   SCORE VARCHAR(10),
   ID_JEU SMALLINT NOT NULL,
   ID_JOUEUR SMALLINT NOT NULL,
   PRIMARY KEY(ID_PARTIE),
   FOREIGN KEY(ID_JEU) REFERENCES jeu(ID_JEU),
   FOREIGN KEY(ID_JOUEUR) REFERENCES joueur(ID_JOUEUR)
);
